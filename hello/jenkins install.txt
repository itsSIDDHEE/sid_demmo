# 1. Install Java (Jenkins needs Java 11 or 17 to run)
sudo apt install openjdk-17-jdk -y

# 2. Clean up the old, broken repository file from your last attempt
sudo rm /etc/apt/sources.list.d/jenkins.list

# 3. Install 'curl' (a tool to download files)
sudo apt install curl -y

# 4. Add the Jenkins GPG key (the modern, correct way)
curl -fsSL https://pkg.jenkins.io/debian-stable/jenkins.io-2023.key | sudo tee \
  /usr/share/keyrings/jenkins-keyring.asc > /dev/null
  
# 5. Add the Jenkins repository, specifying the key location
echo deb [signed-by=/usr/share/keyrings/jenkins-keyring.asc] \
  https://pkg.jenkins.io/debian-stable binary/ | sudo tee \
  /etc/apt/sources.list.d/jenkins.list > /dev/null
  
# 6. Now, update and install Jenkins (this should work!)
sudo apt update
sudo apt install jenkins -y


sudo systemctl start jenkins
sudo systemctl status jenkins
